8/// `TPMI_ALG_HASH`
9#[derive(Clone, Copy, PartialEq, Eq, Debug, Default)]
10#[repr(u16)]
11#[non_exhaustive]
12pub enum TpmiAlgHash {
13    Sha1 = Alg::Sha1.0,
14    #[default]
15    Sha256 = Alg::Sha256.0,
16    Sha384 = Alg::Sha384.0,
17    Sha512 = Alg::Sha512.0,
18    Sm3_256 = Alg::Sm3_256.0,
19    Sha3_256 = Alg::Sha3_256.0,
20    Sha3_384 = Alg::Sha3_384.0,
21    Sha3_512 = Alg::Sha3_512.0,
22}
23use TpmiAlgHash::*;
24
25impl TpmiAlgHash {
26    pub const fn digest_size(self) -> usize {
27        match self {
28            Sha1 => 20,
29            Sha256 => 32,
30            Sha384 => 48,
31            Sha512 => 64,
32            Sm3_256 => 32,
33            Sha3_256 => 32,
34            Sha3_384 => 48,
35            Sha3_512 => 64,
36        }
37    }
38    /// Hash Algs sorted in descending order of digest size
39    pub(crate) const BY_SIZE_DESC: &[Self] = &[
40        Sha512, Sha3_512, Sha384, Sha3_384, Sha256, Sha3_256, Sm3_256, Sha1,
41    ];
42}
43
44impl From<TpmiAlgHash> for Alg {
45    fn from(h: TpmiAlgHash) -> Alg {
46        Alg(h as u16)
47    }
48}
49impl TryFrom<Alg> for TpmiAlgHash {
50    type Error = HashError;
51    fn try_from(a: Alg) -> Result<TpmiAlgHash, Self::Error> {
52        match a {
53            Alg::Sha1 => Ok(Sha1),
54            Alg::Sha256 => Ok(Sha256),
55            Alg::Sha384 => Ok(Sha384),
56            Alg::Sha512 => Ok(Sha512),
57            Alg::Sm3_256 => Ok(Sm3_256),
58            Alg::Sha3_256 => Ok(Sha3_256),
59            Alg::Sha3_384 => Ok(Sha3_384),
60            Alg::Sha3_512 => Ok(Sha3_512),
61            _ => Err(HashError),
62        }
63    }
64}

66impl MarshalFixed for TpmiAlgHash {
67    const SIZE: usize = 2;
68    type Array = [u8; 2];
69    #[inline(always)]
70    fn marshal_fixed(&self, arr: &mut [u8; Self::SIZE]) {
71        Alg::from(*self).marshal_fixed(arr)
72    }
73}
74impl UnmarshalFixed for TpmiAlgHash {
75    fn unmarshal_fixed(&mut self, arr: &[u8; Self::SIZE]) -> Result<(), UnmarshalError> {
76        *self = Alg::unmarshal_value(arr)?.try_into()?;
77        Ok(())
78    }
79}

81/// `TPMT_HA`
82///
83/// There is no type for `TPMU_HA` in this crate, use this type instead.
84#[derive(Copy, Clone, PartialEq, Eq, Debug)]
85#[repr(u16)]
86pub enum TpmtHa<'a> {
87    Sha1(&'a [u8; Sha1.digest_size()]) = Alg::Sha1.0,
88    Sha256(&'a [u8; Sha256.digest_size()]) = Alg::Sha256.0,
89    Sha384(&'a [u8; Sha384.digest_size()]) = Alg::Sha384.0,
90    Sha512(&'a [u8; Sha512.digest_size()]) = Alg::Sha512.0,
91    // TODO: Add other Hash Algs
92}
93
94impl<'a> TpmtHa<'a> {
95    pub const fn hash_alg(self) -> TpmiAlgHash {
96        match self {
97            Self::Sha1(_) => Sha1,
98            Self::Sha256(_) => Sha256,
99            Self::Sha384(_) => Sha384,
100            Self::Sha512(_) => Sha512,
101        }
102    }
103    pub fn digest(self) -> &'a [u8] {
104        match self {
105            Self::Sha1(d) => d,
106            Self::Sha256(d) => d,
107            Self::Sha384(d) => d,
108            Self::Sha512(d) => d,
109        }
110    }
111}
112
113impl<'a> Marshal for TpmtHa<'a> {
114    fn marshal<'dst>(
115        &self,
116        limits: impl Limits,
117        buf: &'dst mut [u8],
118    ) -> Result<&'dst mut [u8], MarshalError> {
119        if !limits.supports_hash(self.hash_alg()) {
120            return Err(MarshalError);
121        }
122        let buf = self.hash_alg().marshal(limits, buf)?;
123        match self {
124            Self::Sha1(d) => d.marshal(limits, buf),
125            Self::Sha256(d) => d.marshal(limits, buf),
126            Self::Sha384(d) => d.marshal(limits, buf),
127            Self::Sha512(d) => d.marshal(limits, buf),
128        }
129    }
130    fn marshaled_size(&self) -> usize {
131        2 + self.hash_alg().digest_size()
132    }
133    fn marshaled_size_max(limits: impl Limits) -> usize {
134        2 + limits.max_digest_size()
135    }
136}
137impl<'a, 'src: 'a> Unmarshal<'src> for TpmtHa<'a> {
138    fn unmarshal(
139        &mut self,
140        limits: impl Limits,
141        buf: &'src [u8],
142    ) -> Result<&'src [u8], UnmarshalError> {
143        let (arr, buf) = buf.split_first_chunk().ok_or(UnmarshalError)?;
144        match TpmiAlgHash::unmarshal_value(arr)? {
145            Sha1 if limits.supports_hash(Sha1) => {
146                let (arr, buf) = buf.split_first_chunk().ok_or(UnmarshalError)?;
147                *self = Self::Sha1(arr);
148                Ok(buf)
149            }
150            Sha256 if limits.supports_hash(Sha256) => {
151                let (arr, buf) = buf.split_first_chunk().ok_or(UnmarshalError)?;
152                *self = Self::Sha256(arr);
153                Ok(buf)
154            }
155            Sha384 if limits.supports_hash(Sha384) => {
156                let (arr, buf) = buf.split_first_chunk().ok_or(UnmarshalError)?;
157                *self = Self::Sha384(arr);
158                Ok(buf)
159            }
160            Sha512 if limits.supports_hash(Sha512) => {
161                let (arr, buf) = buf.split_first_chunk().ok_or(UnmarshalError)?;
162                *self = Self::Sha512(arr);
163                Ok(buf)
164            }
165            _ => Err(UnmarshalError),
166        }
167    }
